# Types

Inertia Server exports TypeScript types for use in your application.

## Import

```typescript
import type {
  InertiaPage,
  PageProps,
} from 'inertia-server';
```

## InertiaPage

The page object passed to your render function.

```typescript
interface InertiaPage<TProps = Record<string, unknown>> {
  component: string;
  props: TProps;
  url: string;
  version: string;
  encryptHistory?: boolean;
  clearHistory?: boolean;
}
```

### Usage

```tsx
import type { InertiaPage } from 'inertia-server';

interface RootProps {
  page: InertiaPage;
}

export default function Root({ page }: RootProps) {
  return (
    <html>
      <body>
        <div id="app" data-page={JSON.stringify(page)} />
      </body>
    </html>
  );
}
```

## PageProps

Extract props type from a page definition.

```typescript
type PageProps<T> = T extends PageBuilder<infer P>
  ? { [K in keyof P]: PropType<P[K]> }
  : never;
```

### Usage

```typescript
import type { PageProps } from 'inertia-server';

const userPage = definePage({
  component: 'User',
  props: {
    name: prop<string>(),
    email: prop<string>(),
  },
});

// Extract the props type
type UserPageProps = PageProps<typeof userPage>;
// { name: string; email: string }
```

## Errors

```typescript
interface Errors {
  [key: string]: string | string[];
}
```

## Flash

```typescript
interface Flash {
  [key: string]: string;
}
```

## SharedData

Type for data shared across all pages:

```typescript
interface SharedData {
  errors?: Errors;
  flash?: Flash;
  auth?: {
    user: User | null;
  };
}
```

### Usage with usePage

```tsx
import { usePage } from '@inertiajs/react';

interface PageData extends SharedData {
  title: string;
}

export default function Component() {
  const { props } = usePage<PageData>();
  
  return (
    <div>
      <h1>{props.title}</h1>
      {props.flash?.success && (
        <Alert>{props.flash.success}</Alert>
      )}
    </div>
  );
}
```
