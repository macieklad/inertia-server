# Quick Start

This guide will help you set up Inertia Server with Elysia and React.

## Prerequisites

- Bun (v1.0+) or Node.js (v18+)
- Basic understanding of Inertia.js concepts

## Step 1: Create a new project

```bash
mkdir my-inertia-app
cd my-inertia-app
bun init -y
```

## Step 2: Install dependencies

```bash
# Core dependencies
bun add inertia-server elysia react react-dom @inertiajs/react

# Dev dependencies
bun add -d typescript vite @vitejs/plugin-react vite-tsconfig-paths
```

## Step 3: Set up your server

Create `src/server.ts`:

```typescript
import { Elysia } from 'elysia';
import { createInertia, prop } from 'inertia-server';
import { elysiaAdapter } from 'inertia-server/elysia';
import { renderToString } from 'react-dom/server';
import Root from './ui/index';

// Create Inertia instance
const { definePage, createHelper } = createInertia({
  version: '1.0.0',
  render: (pageData) => renderToString(<Root page={pageData} />),
});

// Define pages
const homePage = definePage({
  component: 'Home',
  props: {
    title: prop<string>(),
    message: prop<string>(),
  },
});

// Create Inertia plugin
const inertiaPlugin = elysiaAdapter(createHelper);

// Create app
const app = new Elysia()
  .use(inertiaPlugin)
  .get('/', ({ inertia }) => {
    return inertia.render(homePage({
      title: 'Hello World',
      message: 'Welcome to Inertia Server!',
    }));
  })
  .listen(3000);

console.log(`Server running at http://localhost:${app.server?.port}`);
```

## Step 4: Create the root template

Create `src/ui/index.tsx`:

```tsx
import type { InertiaPage } from 'inertia-server';

interface Props {
  page: InertiaPage;
}

export default function Root({ page }: Props) {
  return (
    <html lang="en">
      <head>
        <meta charSet="UTF-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>My Inertia App</title>
      </head>
      <body>
        <div id="app" data-page={JSON.stringify(page)} />
        <script type="module" src="/vite/@vite/client" />
        <script type="module" src="/vite/main.tsx" />
      </body>
    </html>
  );
}
```

## Step 5: Create the client entry

Create `src/ui/main.tsx`:

```tsx
import { createInertiaApp } from '@inertiajs/react';
import { createRoot } from 'react-dom/client';

createInertiaApp({
  resolve: (name) => {
    const pages = import.meta.glob('./pages/**/*.tsx', { eager: true });
    return pages[`./pages/${name}.tsx`];
  },
  setup({ el, App, props }) {
    createRoot(el).render(<App {...props} />);
  },
});
```

## Step 6: Create your first page

Create `src/ui/pages/Home.tsx`:

```tsx
interface Props {
  title: string;
  message: string;
}

export default function Home({ title, message }: Props) {
  return (
    <div>
      <h1>{title}</h1>
      <p>{message}</p>
    </div>
  );
}
```

## Step 7: Run the development server

```bash
bun run --hot src/server.ts
```

Visit [http://localhost:3000](http://localhost:3000) to see your app!
