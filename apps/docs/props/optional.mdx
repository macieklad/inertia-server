# Optional Props

Optional props are not included by default and must be explicitly requested.

## When to Use

- Heavy data that's not always needed
- Secondary content areas
- Data loaded on user action

## Definition

```typescript
import { createInertia, prop } from 'inertia-server';

const { definePage } = createInertia({ version: '1.0.0', render: renderToHtml });

const searchPage = definePage({
  component: 'Search',
  props: {
    // Always included
    query: prop<string>(),
    results: prop<Result[]>(),
    
    // Only when requested
    suggestions: prop<string[]>().optional(),
    relatedSearches: prop<Search[]>().optional(),
  },
});
```

## Usage

```typescript
app.get('/search', ({ inertia }) => {
  return inertia.render(searchPage({
    query: 'test',
    results: await search('test'),
    suggestions: () => getSuggestions('test'),
    relatedSearches: () => getRelatedSearches('test'),
  }));
});
```

## Client-Side Loading

Request optional props explicitly:

```tsx
import { router } from '@inertiajs/react';

export default function Search({ query, results, suggestions }) {
  const loadSuggestions = () => {
    router.reload({ only: ['suggestions'] });
  };

  return (
    <div>
      <h1>Search: {query}</h1>
      <Results data={results} />
      
      {!suggestions ? (
        <button onClick={loadSuggestions}>
          Load Suggestions
        </button>
      ) : (
        <Suggestions data={suggestions} />
      )}
    </div>
  );
}
```

## Combined with Deferred

Optional props can also be deferred:

```typescript
const heavyPage = definePage({
  component: 'Heavy',
  props: {
    // Only loaded when requested, then async
    heavyData: prop<HeavyData>().optional().deferred(),
  },
});
```

## Partial Reloads

Use `only` to request specific optional props:

```tsx
// Load only suggestions
router.reload({ only: ['suggestions'] });

// Load multiple
router.reload({ only: ['suggestions', 'relatedSearches'] });
```
